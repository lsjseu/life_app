```sql

    id BIGINT PRIMARY KEY AUTO_INCREMENT,
```

#### 3.2.3 Agent 完整实现

```python
from langchain_openai import ChatOpenAI
from langchain.agents import AgentExecutor, create_openai_functions_agent
from langchain.memory import ConversationBufferMemory
from langchain_core.runnables.history import RunnableWithMessageHistory

class UnifiedRecordAgent:
    """记录生活Agent"""
    
    def __init__(self, user_id: int, user_profile: Dict[str, Any]):
        self.user_id = user_id
        self.user_profile = user_profile
        
        # 初始化LLM
        self.llm = ChatOpenAI(
            model="gpt-4-turbo-preview",
            temperature=0.7,
            max_tokens=2000
        )
        
        # 初始化工具
        self.tools = [
            food_query_tool,
            exercise_query_tool,
            health_evaluate_tool,
            image_analysis_tool,
            save_record_tool,
        ]
        
        # 初始化记忆
        self.memory = ConversationBufferMemory(
            memory_key="chat_history",
            return_messages=True,
            output_key="output"
        )
        
        # 创建Agent
        self.agent = create_openai_functions_agent(
            llm=self.llm,
            tools=self.tools,
            prompt=RECORD_AGENT_PROMPT
        )
        
        # 创建Agent执行器
        self.agent_executor = AgentExecutor(
            agent=self.agent,
            tools=self.tools,
            memory=self.memory,
            verbose=True,
            return_intermediate_steps=True
        )
    
    async def process(self, input_text: str, images: List[str] = None) -> Dict[str, Any]:
        """处理用户输入"""
        try:
            # 构建输入
            user_input = input_text
            if images:
                user_input += f"\n\n[附件图片: {len(images)}张]"
            
            # 调用Agent
            result = await self.agent_executor.ainvoke({
                "input": user_input,
                "user_profile": self.user_profile,
            })
            
            return {
                "success": True,
                "response": result["output"],
                "intermediate_steps": result.get("intermediate_steps", []),
                "needs_confirmation": self._needs_confirmation(result["output"]),
            }
        except Exception as e:
            return {
                "success": False,
                "error": str(e),
                "response": "抱歉，处理失败了，请稍后重试。"
            }
    
    def _needs_confirmation(self, response: str) -> bool:
        """判断是否需要用户确认"""
        confirmation_keywords = ["对吗", "确认", "是吗", "正确吗"]
        return any(keyword in response for keyword in confirmation_keywords)
    
    async def confirm(self, confirmed: bool, correction: str = None) -> Dict[str, Any]:
        """用户确认"""
        if confirmed:
            return await self.process("对的")
        else:
            if correction:
                return await self.process(f"不对，{correction}")
            else:
                return await self.process("不对")
```

### 3.4 报告生成 Agent 详细设计

#### 3.4.1 Agent 实现

```python
from langchain.chains import LLMChain
from langchain.prompts import PromptTemplate
from datetime import datetime, timedelta

class ReportGeneratorAgent:
    """报告生成Agent"""
    
    def __init__(self, user_id: int):
        self.user_id = user_id
        
        self.llm = ChatOpenAI(
            model="gpt-4-turbo-preview",
            temperature=0.3
        )
    
    async def generate_weekly_report(self, week_start: datetime = None) -> Dict[str, Any]:
        """生成周报"""
        if not week_start:
            # 上周一
            today = datetime.now()
            week_start = today - timedelta(days=today.weekday() + 7)
        
        week_end = week_start + timedelta(days=6)
        
        # 1. 收集数据
        data = await self._collect_data(week_start, week_end, "weekly")
        
        # 2. 分析数据
        analysis = await self._analyze_data(data)
        
        # 3. 生成报告
        report_content = await self._generate_report_content(data, analysis, "weekly")
        
        # 4. 计算健康评分
        score = self._calculate_health_score(data, analysis)
        
        return {
            "type": "weekly",
            "period_start": week_start.date(),
            "period_end": week_end.date(),
            "score": score,
            "content": report_content,
            "data": data,
            "analysis": analysis
        }
    
    async def generate_monthly_report(self, month_start: datetime = None) -> Dict[str, Any]:
        """生成月报"""
        if not month_start:
            # 上个月第一天
            today = datetime.now()
            month_start = (today.replace(day=1) - timedelta(days=1)).replace(day=1)
        
        month_end = (month_start.replace(day=28) + timedelta(days=4)).replace(day=1) - timedelta(days=1)
        
        # 类似周报生成逻辑
        pass
    
    async def _collect_data(self, start_date: datetime, end_date: datetime, report_type: str) -> Dict[str, Any]:
        """收集报告数据"""
        return {
            "diet_records": await self._get_diet_records(start_date, end_date),
            "exercise_records": await self._get_exercise_records(start_date, end_date),
            "health_records": await self._get_health_records(start_date, end_date),
            "user_profile": await self._get_user_profile(),
        }
    
    async def _analyze_data(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """分析数据"""
        analysis_prompt = PromptTemplate.from_template("""
请分析以下健康数据，提供专业的分析：

数据：
{data}

请从以下维度分析：
1. 饮食情况分析
2. 运动情况分析
3. 健康指标趋势
4. 亮点与不足
5. 改进建议

输出格式：JSON
""")
        
        chain = LLMChain(llm=self.llm, prompt=analysis_prompt)
        result = await chain.arun(data=str(data))
        return json.loads(result)
    
    async def _generate_report_content(self, data: Dict[str, Any], analysis: Dict[str, Any], report_type: str) -> Dict[str, Any]:
        """生成报告内容"""
        report_prompt = PromptTemplate.from_template("""
请根据以下数据和分析，生成一份{report_type}健康报告。

数据：
{data}

分析：
{analysis}

报告应该包含：
1. 健康总评
2. 饮食分析
3. 运动分析
4. 健康记录分析
5. 趋势分析
6. 行动建议

用温暖、专业的语言，适合用户阅读。
""")
        
        chain = LLMChain(llm=self.llm, prompt=report_prompt)
        content = await chain.arun(
            report_type=report_type,
            data=str(data),
            analysis=str(analysis)
        )
        
        return {
            "summary": content,
            "sections": self._parse_sections(content)
        }
    
    def _calculate_health_score(self, data: Dict[str, Any], analysis: Dict[str, Any]) -> float:
        """计算健康评分"""
        # 基于多维度计算健康评分
        score = 0.0
        
        # 饮食评分 (30分)
        score += self._calculate_diet_score(data) * 0.3
        
        # 运动评分 (30分)
        score += self._calculate_exercise_score(data) * 0.3
        
        # 健康指标评分 (25分)
        score += self._calculate_health_metrics_score(data) * 0.25
        
        # 记录持续性评分 (15分)
        score += self._calculate_consistency_score(data) * 0.15
        
        return min(100, max(0, score))
```
