## 1. 技术选型

### 1.1 技术栈

| 技术 | 选择 | 说明 |
|-----|------|------|
| 开发框架 | 微信小程序原生 | 轻量化，性能好 |
| 语言 | TypeScript | 类型安全，提升开发效率 |
| 状态管理 | Pinia | 简洁易用，Vue生态 |
| 图表库 | ECharts | 功能强大，支持多种图表 |
| 图片处理 | 微信小程序API | 拍照、相册选择、压缩 |
| 语音识别 | 微信同声传译 | 实时语音转文字 |
| 构建工具 | 微信开发者工具 | 官方工具，调试方便 |

### 1.2 技术选型理由

**微信小程序原生**
- 性能最优，启动速度快
- 官方支持，稳定性好
- 包体积小，用户体验好

**TypeScript**
- 类型安全，减少运行时错误
- 代码提示好，提升开发效率
- 便于维护和重构

**Pinia**
- 轻量级，适合小程序
- API简洁，学习成本低
- TypeScript支持好

```typescript
// components/record/record-overlay/record-overlay.ts
Component({
  properties: {
    visible: {
      type: Boolean,
      value: false,
    },
  },

  data: {
    messages: [],
    conversationId: '',
    isRecording: false,
    isProcessing: false,
    inputText: '',
    selectedImages: [],
  },

  observers: {
    'visible': function(visible) {
      if (visible) {
        this.initConversation();
      }
    },
  },

  methods: {
    initConversation() {
      this.setData({
        messages: [
          {
            id: 'welcome',
            role: 'assistant',
            content: '你好！想记录点什么？\n可以说话、拍照或打字告诉我',
            time: this.formatTime(new Date()),
          },
        ],
        conversationId: this.generateConversationId(),
      });
    },

    async sendMessage(text: string, images: string[] = []) {
      if (!text && images.length === 0) return;

      const userMessage: Message = {
        id: this.generateMessageId(),
        role: 'user',
        content: text,
        images: images,
        time: this.formatTime(new Date()),
      };

      this.setData({
        messages: [...this.data.messages, userMessage],
        isProcessing: true,
        inputText: '',
        selectedImages: [],
      });

      try {
        const response = await api.record.sendMessage({
          conversationId: this.data.conversationId,
          text,
          images,
        });

        const aiMessage: Message = {
          id: this.generateMessageId(),
          role: 'assistant',
          content: response.content,
          quickActions: response.quickActions,
          time: this.formatTime(new Date()),
        };

        this.setData({
          messages: [...this.data.messages, aiMessage],
        });
      } catch (error) {
        wx.showToast({
          title: '发送失败，请重试',
          icon: 'none',
        });
      } finally {
        this.setData({
          isProcessing: false,
        });
      }
    },

    startRecording() {
      this.setData({ isRecording: true });
      voiceUtils.startRecording({
        onResult: (text) => {
          this.setData({ inputText: text });
        },
        onStop: () => {
          this.setData({ isRecording: false });
          if (this.data.inputText) {
            this.sendMessage(this.data.inputText);
          }
        },
      });
    },

    stopRecording() {
      voiceUtils.stopRecording();
    },

    selectImages() {
      imageUtils.chooseImage({
        count: 3,
        success: (images) => {
          this.setData({
            selectedImages: [...this.data.selectedImages, ...images],
          });
        },
      });
    },

    confirmImages() {
      this.sendMessage('', this.data.selectedImages);
    },

    handleQuickAction(action: QuickAction) {
      this.sendMessage(action.text);
    },

    closeOverlay() {
      this.triggerEvent('close');
    },

    minimizeOverlay() {
      this.triggerEvent('minimize');
    },

    generateConversationId() {
      return `conv_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
    },

    generateMessageId() {
      return `msg_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
    },

    formatTime(date: Date) {
      const hours = date.getHours().toString().padStart(2, '0');
      const minutes = date.getMinutes().toString().padStart(2, '0');
      return `${hours}:${minutes}`;
    },
  },
});
```

## 6. API封装

### 6.1 请求封装 (api/request.ts)

```typescript
const BASE_URL = 'https://api.life-app.com';

interface RequestConfig {
  url: string;
  method?: 'GET' | 'POST' | 'PUT' | 'DELETE';
  data?: any;
  header?: any;
}

interface Response<T = any> {
  code: number;
  message: string;
  data: T;
}

class Request {
  private getToken(): string {
    return wx.getStorageSync('token') || '';
  }

  async request<T>(config: RequestConfig): Promise<Response<T>> {
    return new Promise((resolve, reject) => {
      wx.request({
        url: `${BASE_URL}${config.url}`,
        method: config.method || 'GET',
        data: config.data,
        header: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${this.getToken()}`,
          ...config.header,
        },
        success: (res: any) => {
          if (res.statusCode === 200) {
            if (res.data.code === 0) {
              resolve(res.data);
            } else if (res.data.code === 401) {
              wx.removeStorageSync('token');
              wx.navigateTo({ url: '/pages/login/index' });
              reject(new Error('未登录'));
            } else {
              wx.showToast({
                title: res.data.message || '请求失败',
                icon: 'none',
              });
              reject(new Error(res.data.message));
            }
          } else {
            wx.showToast({
              title: '网络错误',
              icon: 'none',
            });
            reject(new Error('网络错误'));
          }
        },
        fail: (err) => {
          wx.showToast({
            title: '网络错误',
            icon: 'none',
          });
          reject(err);
        },
      });
    });
  }

  get<T>(url: string, data?: any) {
    return this.request<T>({ url, method: 'GET', data });
  }

  post<T>(url: string, data?: any) {
    return this.request<T>({ url, method: 'POST', data });
  }

  put<T>(url: string, data?: any) {
    return this.request<T>({ url, method: 'PUT', data });
  }

  delete<T>(url: string, data?: any) {
    return this.request<T>({ url, method: 'DELETE', data });
  }
}

export default new Request();
```

```typescript
            const data = JSON.parse(res.data);
            resolve(data.data.url);
          },
          fail: reject,
        });
      });
    });
    return Promise.all(uploadPromises);
  },
};
```

```typescript
export function formatTime(date: Date): string {
  const hours = date.getHours().toString().padStart(2, '0');
  const minutes = date.getMinutes().toString().padStart(2, '0');
  return `${hours}:${minutes}`;
}

export function formatDate(date: Date): string {
  const year = date.getFullYear();
  const month = (date.getMonth() + 1).toString().padStart(2, '0');
  const day = date.getDate().toString().padStart(2, '0');
  return `${year}-${month}-${day}`;
}

export function formatDateTime(date: Date): string {
  return `${formatDate(date)} ${formatTime(date)}`;
}

export function formatRelativeTime(date: Date): string {
  const now = new Date();
  const diff = now.getTime() - date.getTime();
  const minutes = Math.floor(diff / 60000);
  const hours = Math.floor(diff / 3600000);
  const days = Math.floor(diff / 86400000);

  if (minutes < 1) return '刚刚';
  if (minutes < 60) return `${minutes}分钟前`;
  if (hours < 24) return `${hours}小时前`;
  if (days < 7) return `${days}天前`;
  return formatDate(date);
}

export function formatBMI(bmi: number): string {
  return bmi.toFixed(1);
}

export function getBMIStatus(bmi: number): 'underweight' | 'normal' | 'overweight' | 'obese' {
  if (bmi < 18.5) return 'underweight';
  if (bmi < 24) return 'normal';
  if (bmi < 28) return 'overweight';
  return 'obese';
}

export function getBMIStatusText(status: string): string {
  const texts = {
    underweight: '偏瘦',
    normal: '正常',
    overweight: '偏胖',
    obese: '肥胖',
  };
  return texts[status as keyof typeof texts] || '未知';
```

```typescript
interface VoiceOptions {

  onStop?: () => void;

export default {
  isRecording: false,
  recorderManager: null as any,

  init() {
    this.recorderManager = wx.getRecorderManager();

    this.recorderManager.onStart(() => {
      this.isRecording = true;
    });

    this.recorderManager.onStop((res: any) => {
      this.isRecording = false;
      this.recognizeVoice(res.tempFilePath);
    });

    this.recorderManager.onError((err: any) => {
      console.error('录音错误', err);
    });
  },

  startRecording(options: VoiceOptions = {}) {
    if (!this.recorderManager) {
      this.init();
    }

    this.recorderManager.start({
      duration: 60000,
      sampleRate: 16000,
      numberOfChannels: 1,
      encodeBitRate: 48000,
      format: 'mp3',
    });
  },

  stopRecording() {
    if (this.recorderManager) {
      this.recorderManager.stop();
    }
  },

  async recognizeVoice(filePath: string) {
    try {
      const plugin = requirePlugin('WechatSI');
      const result = await plugin.translate({
        lfrom: 'zh_CN',
        lto: 'zh_CN',
        content: filePath,
        type: 1,
      });

      if (result.retcode === 0) {
        return result.result;
      } else {
        throw new Error(result.msg);
      }
    } catch (error) {
      console.error('语音识别错误', error);
      wx.showToast({
        title: '语音识别失败',
        icon: 'none',
      });
      return '';
    }
  },
};
```

## 8. 开发规范

### 8.1 命名规范

**文件命名**
- 页面文件夹：小写 + 连字符（如：`record-detail`）
- 组件文件夹：小写 + 连字符（如：`message-list`）
- 工具文件：小写 + 连字符（如：`format.ts`）

**变量命名**
- 普通变量：驼峰命名（如：`userInfo`）
- 常量：全大写 + 下划线（如：`MAX_COUNT`）
- 私有变量：下划线开头（如：`_internalValue`）

**函数命名**
- 普通函数：驼峰命名（如：`getUserInfo`）
- 事件处理：`on` + 事件名（如：`onTap`）
- 生命周期：`on` + 生命周期名（如：`onLoad`）

### 8.2 代码规范

**TypeScript使用**
- 所有函数必须添加类型注解
- 使用接口定义对象类型
- 使用类型别名定义联合类型

**组件开发**
- 使用Component语法
- properties必须定义类型
- 使用observers监听属性变化

**状态管理**
- 统一使用Pinia管理状态
- 避免直接修改state，使用actions
- 使用getters计算衍生状态

### 8.3 Git提交规范

```
<type>(<scope>): <subject>

- feat: 新功能
- fix: 修复bug
- docs: 文档更新
```
